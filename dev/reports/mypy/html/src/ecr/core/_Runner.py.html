    Success: int = 0
    Error: int = 1
    TimeOut: int = 2

        self.usedTime: float = 0
        self.isRunning: bool = False
        self.io: str = io
        self.canInput: bool = io[0] == "s"

        self.proc.terminate()

        self.proc.stdin.write(data.encode("utf-8"))
